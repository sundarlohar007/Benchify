    Write `server/src/routes/ws.rs`:

    ```rust
    use axum::extract::ws::{WebSocketUpgrade, WebSocket, Message};
    use axum::extract::{Path, State};
    use axum::response::IntoResponse;
    use futures::{SinkExt, StreamExt};
    use tokio::sync::broadcast;

    pub async fn ws_handler(
        ws: WebSocketUpgrade,
        State(state): State<AppState>,
        Path(session_id): Path<Uuid>,
    ) -> impl IntoResponse {
        ws.on_upgrade(move |socket| handle_socket(socket, state, session_id))
    }

    async fn handle_socket(socket: WebSocket, state: AppState, session_id: Uuid) {
        let (mut sender, mut receiver) = socket.split();

        // Get or create broadcast channel for this session
        let rx = {
            let mut sessions = state.live_sessions.lock().unwrap();
            let tx = sessions.entry(session_id)
                .or_insert_with(|| {
                    let (tx, _) = broadcast::channel(1024); // ring buffer for 1024 samples
                    tx
                });
            tx.subscribe()
        };

        // Spawn task to forward broadcast messages to WebSocket client
        let mut send_task = tokio::spawn(async move {
            let mut rx = rx;
            while let Ok(sample) = rx.recv().await {
                let json = serde_json::to_string(&sample).unwrap();
                if sender.send(Message::Text(json.into())).await.is_err() {
                    break; // client disconnected
                }
            }
        });

        // Spawn task to handle incoming messages (pings, close)
        let recv_task = tokio::spawn(async move {
            while let Some(Ok(msg)) = receiver.next().await {
                match msg {
                    Message::Close(_) => break,
                    Message::Ping(_) => { /* axum handles pong automatically */ }
                    _ => {}
                }
            }
        });

        // Wait for either task to finish
        tokio::select! {
            _ = (&mut send_task) => recv_task.abort(),
            _ = (&mut recv_task) => send_task.abort(),
        };

        tracing::info!(session_id = %session_id, "WebSocket client disconnected");
    }
    ```

    ```rust
    pub async fn push_live_sample(
        State(state): State<AppState>,
        Path(session_id): Path<Uuid>,
        Json(sample): Json<MetricSample>,
    ) -> Result<impl IntoResponse, AppError> {
        // Push sample into broadcast channel
        let sessions = state.live_sessions.lock().unwrap();
        if let Some(tx) = sessions.get(&session_id) {
            let _ = tx.send(sample); // ignore error if no subscribers
            Ok(StatusCode::OK)
        } else {
            // Channel doesn't exist yet — create it or return 404
            Ok(StatusCode::OK) // Silently accept — browsers may connect later
        }
    }
    ```

    ```dart
    import 'dart:async';
    import 'dart:convert';
    import 'package:http/http.dart' as http;
    import '../models/metric_sample.dart';
    import 'api_service.dart';

    class LiveService {
      final ApiService _api;
      Timer? _timer;
      final List<MetricSample> _buffer = [];
      bool _isStreaming = false;

      LiveService(this._api);

      /// Start streaming metric samples to server (5s batch interval).
      /// Call this from ActiveSessionScreen when live mode is enabled.
      void startStreaming(String sessionId) {
        if (_isStreaming) return;
        _isStreaming = true;
        _timer = Timer.periodic(const Duration(seconds: 5), (_) => _flush(sessionId));
      }

      /// Buffer a sample. Called by MetricCollector.onSample callback.
      void pushSample(MetricSample sample) {
        if (!_isStreaming) return;
        _buffer.add(sample);
      }

      Future<void> _flush(String sessionId) async {
        if (_buffer.isEmpty) return;
        final batch = _buffer.toList();
        _buffer.clear();
        try {
          final body = jsonEncode({'samples': batch.map((s) => s.toMap()).toList()});
          await _api.post('/api/v1/sessions/$sessionId/live/batch', jsonDecode(body));
        } catch (e) {
          // Log failure, continue — live streaming is best-effort
          debugPrint('LiveService: failed to flush batch: $e');
        }
      }

      void stopStreaming() {
        _timer?.cancel();
        _timer = null;
        _isStreaming = false;
        _buffer.clear();
      }
    }
    ```

    Write `src/hooks/useWebSocket.ts`:
    ```typescript
    import { useEffect, useRef, useCallback } from 'react';

    export function useWebSocket(sessionId: string | null) {
      const wsRef = useRef<WebSocket | null>(null);
      const reconnectTimeoutRef = useRef<number>();
      const listenersRef = useRef<Set<(data: any) => void>>(new Set());

      const connect = useCallback(() => {
        if (!sessionId) return;
        const protocol = window.location.protocol === 'https:' ? 'wss:' : 'ws:';
        const wsUrl = `${protocol}//${window.location.host}/ws/live/${sessionId}`;
        const ws = new WebSocket(wsUrl);

        ws.onmessage = (event) => {
          const sample = JSON.parse(event.data);
          listenersRef.current.forEach(fn => fn(sample));
        };

        ws.onclose = () => {
          // Auto-reconnect after 2s
          reconnectTimeoutRef.current = window.setTimeout(connect, 2000);
        };

        ws.onerror = () => {
          ws.close();
        };

        wsRef.current = ws;
      }, [sessionId]);

      useEffect(() => {
        connect();
        return () => {
          wsRef.current?.close();
          if (reconnectTimeoutRef.current) clearTimeout(reconnectTimeoutRef.current);
        };
      }, [connect]);

      const onSample = useCallback((fn: (sample: MetricSample) => void) => {
        listenersRef.current.add(fn);
        return () => { listenersRef.current.delete(fn); };
      }, []);

      return { onSample };
    }
    ```

    ```typescript
    export function LiveChart({ sessionId, metric, color }: LiveChartProps) {
      const chartRef = useRef<Chart | null>(null);
      const canvasRef = useRef<HTMLCanvasElement>(null);
      const { onSample } = useWebSocket(sessionId);

      // Initialize Chart.js instance once
      useEffect(() => {
        if (!canvasRef.current) return;
        const ctx = canvasRef.current.getContext('2d')!;
        chartRef.current = new Chart(ctx, {
          type: 'line',
          data: { datasets: [{ label: metric, data: [], borderColor: color, pointRadius: 0 }] },
          options: { ...darkThemeOptions, animation: { duration: 0 } }, // no animation for live
        });
        return () => { chartRef.current?.destroy(); };
      }, [metric, color]);

      // Append data points as they arrive
      useEffect(() => {
        return onSample((sample) => {
          if (!chartRef.current) return;

          chartRef.current.data.datasets[0].data.push({ x: time, y: value });
          // Keep only last 300 points (5 minutes at 1Hz)

            chartRef.current.data.datasets[0].data.shift();
          }
          chartRef.current.update('none'); // 'none' mode = no animation
        });
      }, [onSample, metric]);

      return <canvas ref={canvasRef} />;
    }
    ```

| Browser → Server (WebSocket /ws/live/:id) | Untrusted: WebSocket upgrade from browser; no auth on upgrade handshake (session ID is URL param) |
| Desktop → Server (POST /api/v1/sessions/:id/live/batch) | Untrusted: API token Bearer auth on live push endpoint |
| Mobile App → Server (REST API) | Untrusted: API token Bearer auth, same as desktop |
| Server broadcast channel | Internal: tokio::sync::broadcast fan-out to WebSocket clients |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |

| T-03-32 | Information Disclosure | live_session broadcast — cross-user data leak | mitigate | Session ID is a UUID v4 (unguessable); broadcast channel per session_id; no way to subscribe to another user's session without knowing the UUID; rate-limit subscription attempts |
