    **BenchifyBroadcastReceiver.java** — Android BroadcastReceiver:
    ```java
    package dev.benchify;

    import android.content.BroadcastReceiver;
    import android.content.Context;
    import android.content.Intent;
    import android.os.Bundle;

    public class BenchifyBroadcastReceiver extends BroadcastReceiver {
        private static final String ACTION_BROADCAST = "com.benchify.COMMAND";
        private static final String RESPONSE_BROADCAST = "com.benchify.RESPONSE";

        // JNI: native handler returns JSON response string
        public static native String nativeHandleCommand(String action, String payloadJson);

        @Override
        public void onReceive(Context context, Intent intent) {
            if (!ACTION_BROADCAST.equals(intent.getAction())) return;

            Bundle extras = intent.getExtras();
            String action = extras != null ? extras.getString("action", "") : "";
            String payload = extras != null ? extras.getString("payload", "{}") : "{}";

            // Dispatch to Rust native handler
            String responseJson = nativeHandleCommand(action, payload);

            // Send response broadcast
            Intent responseIntent = new Intent(RESPONSE_BROADCAST);
            responseIntent.putExtra("status", responseJson);
            context.sendBroadcast(responseIntent);
        }
    }
    ```

    **automation.rs** — Rust command handler module:
    - Module `automation` with function `handle_command(action: &str, payload_json: &str) -> String`.
    - Dispatches on action string (per D-22):
      1. **START_SESSION** — Begins metric collection + TCP streaming. Payload: `{"session_id": "abc-123"}`. Response: `{"action":"START_SESSION","status":"ok","detail":"Profiling started","session_id":"abc-123"}`
      2. **STOP_SESSION** — Stops metric collection, closes TCP connections, flushes buffers. Payload: `{}`. Response: `{"action":"STOP_SESSION","status":"ok","detail":"Profiling stopped","session_id":"abc-123"}`
      3. **PAUSE** — Pauses metric collection (TCP remains open, no new samples). Payload: `{}`. Response: `{"action":"PAUSE","status":"ok"}`
      4. **RESUME** — Resumes paused metric collection. Payload: `{}`. Response: `{"action":"RESUME","status":"ok"}`
      5. **MARKER** — Inserts a session marker at current timestamp. Payload: `{"note": "boss fight start"}`. Response: `{"action":"MARKER","status":"ok","marker_id":42,"note":"boss fight start","timestamp_ms":1700000000000}`
      6. **SCREENSHOT** — Captures current screen via `SurfaceControl.screenshot()` or similar, writes PNG to device storage. Payload: `{"label": "death_screen"}`. Response: `{"action":"SCREENSHOT","status":"ok","path":"/sdcard/benchify/abc-123_death_screen.png"}`
      7. **EXPORT** — Triggers session data export: writes accumulated MetricSample buffer to JSON file on device storage. Payload: `{}`. Response: `{"action":"EXPORT","status":"ok","path":"/sdcard/benchify/abc-123_export.json","sample_count":360}`

    - All handlers use shared state (Arc<Mutex<SdkState>>) to access the metric collector and TCP transport.
    - Error responses: `{"action":"...","status":"error","detail":"reason string"}`

    **dvt_recorder.py** — Python 3.10+ script using pymobiledevice3 DVT:
    ```python
    #!/usr/bin/env python3
    """
    iOS DVT screen-mirror recorder — pymobiledevice3 subprocess.
    Usage: python3 dvt_recorder.py <udid> [--quality 1080p|720p|480p] [--fps 30|60|15] [--output-dir /tmp]

    Streams DVT screen data, pipes to ffmpeg for H.264 MP4 encoding.
    Auto-chunks at 5-minute intervals (same as Android ScreenrecordService).
    Emits JSON status lines to stdout for desktop parsing.
    """
    import sys, json, time, subprocess, signal, os, argparse
    from pymobiledevice3.services.dvt.dvt_secure_socket_proxy import DvtSecureSocketProxyService

    def main():
        parser = argparse.ArgumentParser()
        parser.add_argument('udid')
        parser.add_argument('--quality', default='1080p', choices=['480p','720p','1080p'])
        parser.add_argument('--fps', type=int, default=30, choices=[15,30,60])
        parser.add_argument('--output-dir', default='data/videos')
        parser.add_argument('--session-id', required=True)
        args = parser.parse_args()

        # Quality → resolution mapping per D-20
        resolutions = {'480p': (640,480), '720p': (1280,720), '1080p': (1920,1080)}
        width, height = resolutions[args.quality]
        fps = args.fps

        # Connect DVT service via pymobiledevice3
        stopped = False
        def on_sigterm(signum, frame):
            nonlocal stopped
            stopped = True
        signal.signal(signal.SIGTERM, on_sigterm)

        # Emit started status
        print(json.dumps({"event":"recording_started","session_id":args.session_id,
                         "width":width,"height":height,"fps":fps}))
        sys.stdout.flush()

        chunk_index = 0
        chunk_duration = 300  # 5 minutes per D-17

        while not stopped:
            chunk_index += 1
            chunk_file = f"{args.session_id}_chunk_{chunk_index:03d}.mp4"
            chunk_path = os.path.join(args.output_dir, args.session_id, chunk_file)
            os.makedirs(os.path.dirname(chunk_path), exist_ok=True)

            chunk_start = int(time.time() * 1000)
            print(json.dumps({"event":"chunk_start","chunk":chunk_index,
                             "file":chunk_file,"timestamp_ms":chunk_start}))
            sys.stdout.flush()

            # ffmpeg pipe: DVT raw frames → H.264 MP4
            # DvtSecureSocketProxyService provides screen frames
            # Pipe to ffmpeg for encoding
            ffmpeg_cmd = [
                'ffmpeg', '-y',
                '-f', 'rawvideo',
                '-pixel_format', 'bgra',
                '-video_size', f'{width}x{height}',
                '-framerate', str(fps),
                '-i', 'pipe:0',        # stdin from DVT frames
                '-c:v', 'libx264',
                '-preset', 'ultrafast',
                '-crf', '23',
                '-pix_fmt', 'yuv420p',
                '-an',                 # No audio per D-21
                '-t', str(chunk_duration),
                chunk_path
            ]

            proc = subprocess.Popen(ffmpeg_cmd, stdin=subprocess.PIPE, stderr=subprocess.DEVNULL)

            try:
                # Connect DVT and stream frames to ffmpeg stdin
                # DvtSecureSocketProxyService provides frame callbacks
                # Each frame written to proc.stdin as raw BGRA bytes
                with DvtSecureSocketProxyService(lockdown=lockdown_client) as dvt:
                    def frame_callback(frame_data):
                        if stopped:
                            return False  # Stop streaming
                        try:
                            proc.stdin.write(frame_data)
                        except BrokenPipeError:
                            return False
                        return True

                    dvt.start_screen_mirror(frame_callback, fps=fps)
            except Exception as e:
                print(json.dumps({"event":"chunk_error","chunk":chunk_index,
                                 "error":str(e)}))
                sys.stdout.flush()
            finally:
                proc.stdin.close()
                proc.wait(timeout=5)

            chunk_end = int(time.time() * 1000)
            file_size = os.path.getsize(chunk_path) if os.path.exists(chunk_path) else 0
            print(json.dumps({"event":"chunk_end","chunk":chunk_index,
                             "file":chunk_file,"timestamp_ms":chunk_end,
                             "size_bytes":file_size}))
            sys.stdout.flush()

        print(json.dumps({"event":"recording_stopped","session_id":args.session_id}))
        sys.stdout.flush()

    if __name__ == '__main__':
        main()
    ```

      final String _python3Path;
      final String _agentsDir;
      final VideoDao? _videoDao;
      final String _dataDir;
      Process? _process;
      String? _sessionId;
      int _chunkIndex = 0;
      final List<_IosChunkRecord> _chunks = [];
      Timer? _chunkTimer;
      int _width = 1080;
      int _height = 1920;
      int _fps = 30;

      bool get isRecording => _sessionId != null;

      /// Guards: only macOS supports this. Throws StateError otherwise.
      static void checkPlatform() {
        if (!Platform.isMacOS) {
          throw StateError('iOS video recording requires macOS host');
        }
      }

      void configure({int width = 1080, int height = 1920, int fps = 30}) { ... }

      /// Start DVT screen recording via pymobiledevice3.
      /// Spawns dvt_recorder.py subprocess. Returns true on success.
      Future<bool> start({required String sessionId, required String udid}) async {
        checkPlatform();
        if (isRecording) return false;
        _sessionId = sessionId;
        _chunks.clear();

        _spawnDvtRecorder(udid, sessionId);
        return true;
      }

      Future<void> _spawnDvtRecorder(String udid, String sessionId) async {
        _process = await Process.start(
          _python3Path,
          [
            '$_agentsDir/dvt_recorder.py', udid,
            '--quality', _qualityPreset,
            '--fps', _fps.toString(),
            '--output-dir', p.join(_dataDir, 'videos'),
            '--session-id', sessionId,
          ],
        );
        _process!.stdout
          .transform(utf8.decoder)
          .transform(const LineSplitter())
          .listen(_onDvtLine);
        _process!.stderr.transform(utf8.decoder).listen((line) {
          // Log stderr for diagnostics
          // ignore: avoid_print
          print('[ios_screenrecord stderr] $line');
        });
      }

      void _onDvtLine(String line) {
        try {
          final json = jsonDecode(line) as Map<String, dynamic>;
          switch (json['event']) {
            case 'chunk_start':
              _chunks.add(_IosChunkRecord(
                chunkIndex: json['chunk'] as int,
                fileName: json['file'] as String,
                startMs: json['timestamp_ms'] as int,
              ));
              break;
            case 'recording_stopped':
              // Already handled by stop()
              break;
            case 'chunk_error':
              // Log error, continue
              break;
          }
        } catch (_) { /* skip malformed lines */ }
      }

      /// Stop recording — SIGTERM → 3s → SIGKILL.
      /// Pulls chunk metadata, builds Video model, saves via VideoDao.
      Future<Video?> stop() async {
        if (!isRecording || _sessionId == null) return null;
        // Cancel any pending chunk timer
        _chunkTimer?.cancel();
        _process?.kill(ProcessSignal.sigterm);
        await Future.delayed(const Duration(seconds: 3));
        if (_process != null) {
          _process!.kill(ProcessSignal.sigkill);
          _process = null;
        }

        // Build Video model matching ScreenrecordService pattern exactly
        if (_chunks.isEmpty) return null;
        final chunks = _chunks.map((c) => {
          'chunk': c.chunkIndex,
          'file': c.fileName,
          'startMs': c.startMs,
        }).toList();
        final primaryFile = _chunks.first.fileName;
        final totalDuration = _chunks.length * 300000; // 5 min chunks

        final video = Video(
          sessionId: _sessionId!,
          filepath: p.join(_dataDir, 'videos', _sessionId!, primaryFile),
          codec: 'h264',
          container: 'mp4',
          widthPx: _width,
          heightPx: _height,
          targetFps: _fps,
          actualAvgFps: null,
          bitrateKbps: _bitrate,
          durationMs: totalDuration,
          fileSizeBytes: 0, // Updated after file stat
          chunksJson: jsonEncode(chunks),
          gapsJson: null,
          hasAudio: 0,  // Per D-21: video-only
          recordingOverheadEstimatePct: 5.0,
          startedAt: _chunks.first.startMs,
          endedAt: DateTime.now().millisecondsSinceEpoch,
          createdAt: DateTime.now().millisecondsSinceEpoch,
        );
        if (_videoDao != null) await _videoDao!.insert(video);

        _sessionId = null;
        _chunks.clear();
        return video;
      }

      /// Abort recording without saving.
      Future<void> abort() async {
        _chunkTimer?.cancel();
        if (_process != null) {
          _process!.kill(ProcessSignal.sigterm);
          await Future.delayed(const Duration(seconds: 3));
          _process!.kill(ProcessSignal.sigkill);
          _process = null;
        }
        _sessionId = null;
        _chunks.clear();
      }
    }

      final String fileName;
      final int startMs;
      const _IosChunkRecord({
        required this.chunkIndex,
        required this.fileName,
        required this.startMs,
      });
